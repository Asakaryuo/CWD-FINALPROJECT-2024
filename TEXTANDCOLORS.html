TEXT AND COLORS


Text and Colords in TFT Displays are easy to come by and manageable but comes with a little bit of patience. Here's what you'll need:

Code Format Example
MCUFRIEND_kbv MCUFRIEND_kbv tft;
#define #define "Color Callout name" "Hexadecimal Color code" #define "OLIVE" "0x7BE0"
tft.setTextSize tft.setTextSize(font size) tft.setTextSize(4)
tft.setTextColor tft.setTextColor(font Color) tft.setTextColor(BLUE)
tft.setCursor tft.setCursor(x pos,y pos) tft.setCursor(20,200)
tft.print tft.print(text) tft.print(Hello!)


Here is an example code of using the codes above, try to copy it and observe the usage of fonts and colors while manipulating the X and Y coordinates around the TFT Display field. 

            
//TEXT AND COLORS
//Libraries start with #include <'Library.h'>
#include <'Adafruit_GFX.h'>
#include <'MCUFRIEND_kbv.h'>
#include <'TouchScreen.h'>

MCUFRIEND_kbv tft;

//Define the Minimum and Maximum resistance for the Touchscreen threshold
#define MINPRESSURE 100
#define MAXPRESSURE 1000


//This is the list of the color palletes 
#define BLACK       0x0000      /*   0,   0,   0 */
#define NAVY        0x000F      /*   0,   0, 128 */
#define DARKGREEN   0x03E0      /*   0, 128,   0 */
#define DARKCYAN    0x03EF      /*   0, 128, 128 */
#define MAROON      0x7800      /* 128,   0,   0 */
#define PURPLE      0x780F      /* 128,   0, 128 */
#define OLIVE       0x7BE0      /* 128, 128,   0 */
#define LIGHTGREY   0xC618      /* 192, 192, 192 */
#define DARKGREY    0x7BEF      /* 128, 128, 128 */
#define BLUE        0x001F      /*   0,   0, 255 */
#define GREEN       0x07E0      /*   0, 255,   0 */
#define CYAN        0x07FF      /*   0, 255, 255 */
#define RED         0xF800      /* 255,   0,   0 */
#define MAGENTA     0xF81F      /* 255,   0, 255 */
#define YELLOW      0xFFE0      /* 255, 255,   0 */
#define WHITE       0xFFFF      /* 255, 255, 255 */
#define ORANGE      0xFD20      /* 255, 165,   0 */
#define GREENYELLOW 0xAFE5      /* 173, 255,  47 */
#define PINK        0xF81F

//This is for the Call out Pins on the Arduino Pins
const int XP = 6, XM = A2, YP = A1, YM = 7;
//This is for the sensitivity of the Touchscreen Module
//NOTE!!! ~~ This differs on TFT Modules check it with the callibration
const int TS_LEFT =907, TS_RT = 136, TS_TOP = 942, TS_BOT = 139;


void setup() {
// put your setup code here, to run once:
//This is used to read the driver chips inside the TFT for it to run smoothly
uint16_t ID = tft.readID(); 
if (ID == 0xD3D3) ID = 0x9486; //
tft.begin(ID);    //Use this to Initialize the TFT
tft.fillScreen(BLACK);  //This is normally used to restart the whole Screen
}

void loop() {
// put your main code here, to run repeatedly:

tft.setTextSize(2);
tft.setTextColor(RED);
tft.setCursor(110,10);
tft.print("HELLO");

tft.setTextSize(4);
tft.setTextColor(BLUE);
tft.setCursor(110,30);
tft.print("HELLO");

tft.setTextSize(10);
tft.setTextColor(WHITE);
tft.setCursor(20,90);
tft.print("HELLO");

tft.setTextSize(10);
tft.setTextColor(WHITE);
tft.setCursor(20,200);
tft.print("HELLO");

tft.setTextSize(10);
tft.setTextColor(WHITE);
tft.setCursor(20,320);
tft.print("HELLO");

}
This is what your TFT Display should be looking right now! You can always change the colors just remember putting the hexacimal color code and set up its name and you're good to go.
